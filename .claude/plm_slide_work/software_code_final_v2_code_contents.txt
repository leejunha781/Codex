"""Optional PostgreSQL persistence for AI Gate evaluations.

The FastAPI app can run without a database for local tests.  When DATABASE_URL is
set and PostgreSQL is reachable, the gateway writes evaluation, finding and
override audit rows into the SQL schema defined in sql/init/04_ai_gate.sql.
"""

import json
import os
from typing import Any

from app.domain.ai_models import AIGateEvaluation, AIGateOverrideResult


class PostgresAIGateRepository:
    def __init__(self, dsn: str | None = None) -> None:
        self.dsn = dsn or os.getenv("DATABASE_URL", "")

    @property
    def enabled(self) -> bool:
        return bool(self.dsn)

    async def _connect(self):
        import psycopg
        return await psycopg.AsyncConnection.connect(self.dsn)

    async def save_evaluation(self, evaluation: AIGateEvaluation, context_payload: dict[str, Any], evidence_refs: list[str]) -> None:
        if not self.enabled:
            return
        try:
            async with await self._connect() as conn:
                async with conn.cursor() as cur:
                    await cur.execute(
                        """
                        INSERT INTO plm_ai_gate_evaluation(
                            evaluation_id, project_id, gate_type, object_type, object_id,
                            provider_code, decision, confidence, human_release_required,
                            release_authority, prompt_hash, response_hash, context_payload,
                            evidence_refs, guardrail, created_at
                        ) VALUES (%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s::jsonb,%s::jsonb,%s,%s)
                        ON CONFLICT (evaluation_id) DO NOTHING;
                        """,
                        (
                            evaluation.evaluation_id,
                            evaluation.project_id,
                            str(evaluation.gate_type.value if hasattr(evaluation.gate_type, "value") else evaluation.gate_type),
                            evaluation.object_type,
                            evaluation.object_id,
                            evaluation.provider_code,
                            str(evaluation.decision.value if hasattr(evaluation.decision, "value") else evaluation.decision),
                            evaluation.confidence,
                            evaluation.human_release_required,
                            evaluation.release_authority,
                            evaluation.prompt_hash,
                            evaluation.response_hash,
                            json.dumps(context_payload),
                            json.dumps(evidence_refs),
                            evaluation.guardrail,
                            evaluation.created_at,
                        ),
                    )
                    for finding in evaluation.findings:
                        await cur.execute(
                            """
                            INSERT INTO plm_ai_gate_finding(
                                finding_id, evaluation_id, severity, category, message,
                                affected_object_type, affected_object_id, evidence_required,
                                recommendation, confidence, ai_generated, created_at
                            ) VALUES (%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s)
                            ON CONFLICT (finding_id) DO NOTHING;
                            """,
                            (
                                finding.finding_id,
                                evaluation.evaluation_id,
                                str(finding.severity.value if hasattr(finding.severity, "value") else finding.severity),
                                finding.category,
                                finding.message,
                                finding.affected_object_type,
                                finding.affected_object_id,
                                finding.evidence_required,
                                finding.recommendation,
                                finding.confidence,
                                finding.ai_generated,
                                evaluation.created_at,
                            ),
                        )
                    await conn.commit()
        except Exception:
            # Do not block gate evaluation if audit persistence is temporarily down.
            return

"""AVEVA CONNECT Industrial AI provider adapter.

This adapter is intentionally written as a thin, replaceable boundary.  AVEVA
CONNECT tenant URLs, authentication method, and production payload contract vary
by customer environment, so the code uses environment variables and falls back to

class AvevaConnectIndustrialAIProvider(AIGateProvider):
    """Provider for AVEVA CONNECT Industrial AI Assistant style evaluation.

    Design intent from the final presentation:
    - CONNECT Industrial AI supports role-gated industrial Q&A and diagnosis.
    - FastAPI remains the PLM control-plane authority.
    - PostgreSQL stores evaluation, evidence reference and audit metadata.
    - Deterministic rules can only tighten an AI decision.
    - AI never owns certified release authority.
    """

    provider_code = "AVEVA_CONNECT_INDUSTRIAL_AI"

    def __init__(self) -> None:
        self.endpoint = os.getenv("AVEVA_CONNECT_AI_ENDPOINT", "").rstrip("/")
        self.api_key = os.getenv("AVEVA_CONNECT_AI_API_KEY", "")
        self.bearer_token = os.getenv("AVEVA_CONNECT_AI_BEARER_TOKEN", "")
        self.model = os.getenv("AVEVA_CONNECT_AI_MODEL", "industrial-ai-assistant")
        self.timeout_seconds = float(os.getenv("AVEVA_CONNECT_AI_TIMEOUT", "30"))
        self.fallback = RuleBasedGuardrailProvider()

    @property
    def configured(self) -> bool:
        return bool(self.endpoint and (self.api_key or self.bearer_token))

    async def evaluate(self, request: AIGateEvaluateRequest) -> AIGateEvaluation:
        prompt = self._build_prompt(request)

from __future__ import annotations

import hashlib
import json

from app.ai.base import AIGateProvider
from app.domain.ai_models import (
    AIFindingSeverity,
    AIGateDecision,
    AIGateEvaluateRequest,
    AIGateEvaluation,
    AIGateFinding,
    AIGateType,
    new_ai_id,
)


class RuleBasedGuardrailProvider(AIGateProvider):
    """Deterministic guardrail provider used when Azure OpenAI is unavailable.

    This is not intended to replace Microsoft AI. It provides predictable gate
    behavior for local tests, customer demos, and fail-safe operation.
    """

    provider_code = "LOCAL_RULES_GUARDRAIL"

    async def evaluate(self, request: AIGateEvaluateRequest) -> AIGateEvaluation:
        findings: list[AIGateFinding] = []
        ctx = request.context_payload or {}
        evidence_refs = request.evidence_refs or []

        if request.gate_type in {AIGateType.ASSURANCE_TO_APPROVAL, AIGateType.CHANGE_TO_ASSURANCE}:
            if not evidence_refs and not ctx.get("solver_run_id"):
                findings.append(AIGateFinding(
                    finding_id=new_ai_id("AIF"),
                    severity=AIFindingSeverity.CRITICAL,
                    category="MISSING_ASSURANCE_EVIDENCE",
                    message="No solver run or evidence reference is attached for an assurance-critical gate.",
                    affected_object_type=request.object_type,
                    affected_object_id=request.object_id,
                    evidence_required="Approved solver run, margin result, rule trace and evidence pack.",
                    recommendation="Run an approved solver adapter and generate an evidence pack before approval.",
                    confidence=0.94,
                ))

        if request.gate_type in {AIGateType.DESIGN_TO_DRAWING, AIGateType.DRAWING_TO_MTO}:
            if not ctx.get("drawing_standard_checked") and not ctx.get("mto_extracted"):
                findings.append(AIGateFinding(
                    finding_id=new_ai_id("AIF"),
                    severity=AIFindingSeverity.HIGH,
                    category="DESIGN_OUTPUT_NOT_VERIFIED",
                    message="The gate context does not confirm drawing-standard check or MTO extraction.",
                    affected_object_type=request.object_type,
                    affected_object_id=request.object_id,
                    evidence_required="Drawing standard report and MTO extraction evidence.",
                    recommendation="Attach drawing standard check result and MTO extraction result.",
                    confidence=0.86,
                ))

        if request.gate_type == AIGateType.REQUIREMENT_TO_DESIGN:
            if not ctx.get("vcrm_items") and not ctx.get("checklist_items"):
                findings.append(AIGateFinding(
                    finding_id=new_ai_id("AIF"),
                    severity=AIFindingSeverity.HIGH,
                    category="MISSING_REQUIREMENT_TRACEABILITY",
                    message="No VCRM or checklist items were provided for this requirement gate.",
                    affected_object_type=request.object_type,
                    affected_object_id=request.object_id,
                    evidence_required="VCRM and checklist generated from requirement.",
                    recommendation="Generate VCRM and checklist before releasing design work.",
                    confidence=0.88,
                ))

        if request.gate_type == AIGateType.APPROVAL_TO_PRODUCTION:
            required = ["vcrm_complete", "checklist_complete", "mbom_released", "mco_closed", "approval_complete"]
            missing = [x for x in required if not ctx.get(x)]
            if missing:
                findings.append(AIGateFinding(
                    finding_id=new_ai_id("AIF"),
                    severity=AIFindingSeverity.CRITICAL,
                    category="PRODUCTION_RELEASE_BLOCKED",
                    message=f"Production release prerequisites are missing: {', '.join(missing)}.",
                    affected_object_type=request.object_type,
                    affected_object_id=request.object_id,
                    evidence_required="Completed VCRM, checklist, M-BOM, MCO and final approval.",
                    recommendation="Close all release prerequisites before production release.",
                    confidence=0.96,
                ))

        if not findings:
            findings.append(AIGateFinding(
                finding_id=new_ai_id("AIF"),
                severity=AIFindingSeverity.INFO,
                category="NO_BLOCKING_ISSUE_FOUND",
                message="No blocking issue was found by deterministic guardrail checks.",
                affected_object_type=request.object_type,
                affected_object_id=request.object_id,
                recommendation="Proceed to human review or approval according to the process policy.",
                confidence=0.80,
            ))

        max_sev = max([f.severity for f in findings], key=lambda s: ["INFO", "LOW", "MEDIUM", "HIGH", "CRITICAL"].index(s.value if hasattr(s, "value") else str(s)))
        if max_sev == AIFindingSeverity.CRITICAL:
            decision = AIGateDecision.HOLD
            confidence = 0.92
        elif max_sev == AIFindingSeverity.HIGH:
            decision = AIGateDecision.REVIEW
            confidence = 0.84
        else:
            decision = AIGateDecision.PASS
            confidence = 0.80

        raw = json.dumps(request.model_dump(mode="json"), sort_keys=True)
        return AIGateEvaluation(
            evaluation_id=new_ai_id("AIG"),
            project_id=request.project_id,
            gate_type=request.gate_type,
            object_type=request.object_type,
            object_id=request.object_id,
            provider_code=self.provider_code,
            decision=decision,
            confidence=confidence,
            findings=findings,
            human_release_required=True,
            release_authority=["ENGINEERING_APPROVER", "PRINCIPAL_CONSULTANT"],
            prompt_hash=hashlib.sha256(raw.encode()).hexdigest(),
            response_hash=hashlib.sha256((raw + str(decision)).encode()).hexdigest(),
        )

from __future__ import annotations

from dataclasses import dataclass
from fastapi import Header, HTTPException, status


@dataclass(frozen=True)
class CurrentUser:
    username: str
    roles: list[str]


def get_current_user(authorization: str | None = Header(default=None)) -> CurrentUser:
    """Lightweight placeholder for local reference mode.

    Production mode should validate JWT/OIDC with JWKS and map Keycloak roles.
    For the executable reference, a missing token maps to a demo admin so local
    Swagger usage remains frictionless. CI can still test RBAC paths explicitly.
    """
    if authorization and authorization.startswith("Bearer "):
        return CurrentUser(username="jwt-user", roles=["plm_admin", "gate_operator", "naval_architect"])
    return CurrentUser(username="demo-admin", roles=["plm_admin", "gate_operator", "naval_architect"])


def require_role(user: CurrentUser, allowed: set[str]) -> None:
    if not allowed.intersection(user.roles):
        raise HTTPException(status_code=status.HTTP_403_FORBIDDEN, detail=f"Required role: {sorted(allowed)}")

class E3DBridgeAdapter(ExternalProgramAdapter):
    """Bridge for AVEVA E3D/Marine/Hull Design.

    Important integration assumption:
    - E3D/Marine/Hull authoring is Windows-centered and must run in its
      supported AVEVA engineering authoring environment.
    - The Linux FastAPI server does not host or directly manipulate E3D internal
      authoring databases. It is the Open PLM Control Plane.
    - A Windows-side E3D plugin / PML.NET customization / local agent exposes a
      safe command contract and posts callbacks back to this API.
    - This protects design-tool performance while still making E3D/Hull revision
      events part of the governed AVEVA PLM control plane.
    """

    def __init__(self, program_code: ProgramCode = ProgramCode.AVEVA_E3D) -> None:
        self.program_code = program_code
        self.integration_mode = IntegrationMode.IN_PRODUCT_PLUGIN

    async def execute_command(self, request: ExternalCommandRequest) -> ExternalCommandResponse:
        command_id = new_integration_id("CMD")
        supported = {
            "extract_mto",
            "export_model_metadata",
            "capture_revision_delta",
            "create_baseline_snapshot",
            "export_geometry_reference",
            "validate_design_rule",
        }
        if request.command_name not in supported:
            return ExternalCommandResponse(
                command_id=command_id,
                program_code=request.program_code,
                command_name=request.command_name,
                status=CommandStatus.REJECTED,
                accepted_at=datetime.utcnow(),
                adapter_mode=self.integration_mode,
                callback_expected=False,
                message=f"Unsupported E3D bridge command. Supported: {sorted(supported)}",
            )
        # In production, enqueue command to a Windows-side E3D bridge/agent: Redis/RabbitMQ/HTTP/gRPC.
        return ExternalCommandResponse(
            command_id=command_id,
            program_code=request.program_code,
            command_name=request.command_name,
            status=CommandStatus.ACCEPTED,
            accepted_at=datetime.utcnow(),
            adapter_mode=self.integration_mode,
            callback_expected=True,
            result={
                "bridge_contract": "E3D_PLUGIN_COMMAND_V1",
                "next_callback": f"/api/v1/integrations/callbacks/{request.program_code}",
                "pml_net_hint": "Windows-side E3D plugin should map command to PML/PML.NET or .NET customization and return metadata delta, not whole 3D files.",
                "deployment_model": "Windows Authoring + Linux Control Plane + Callback Bridge",
                "payload_echo": request.payload,
            },
            message="Command accepted for Windows-side E3D/Hull local bridge execution; callback expected by the Linux Control Plane.",
        )

    async def accept_callback(self, request: ExternalCallbackRequest) -> ExternalCallbackResponse:
        mapped = {
            CallbackEventType.DESIGN_OBJECT_CHANGED: "revision_delta_candidate",
            CallbackEventType.DESIGN_REVISION_PROMOTED: "e3d_revision_event",
            CallbackEventType.BASELINE_FROZEN: "baseline_freeze_event",
            CallbackEventType.MTO_EXTRACTED: "mto_evidence_event",
            CallbackEventType.DRAWING_ISSUED: "drawing_evidence_event",
        }.get(request.event_type, "e3d_generic_event")
        return ExternalCallbackResponse(
            callback_id=new_integration_id("CB"),
            program_code=self.program_code,
            event_type=request.event_type,
            accepted_at=datetime.utcnow(),
            mapped_control_plane_event=mapped,
            message="E3D/Hull callback accepted. Production implementation should map this payload into revision, BOM, evidence or quality gate tables.",
        )

from __future__ import annotations

from datetime import datetime
from typing import Any

import httpx

from app.domain.integration_models import CommandStatus, ExternalCommandRequest, ExternalCommandResponse, IntegrationMode, ProgramCode, new_integration_id
from app.integration.base import ExternalProgramAdapter


class GenericRestAdapter(ExternalProgramAdapter):
    def __init__(self, program_code: ProgramCode, base_url: str | None = None, token: str | None = None) -> None:
        self.program_code = program_code
        self.integration_mode = IntegrationMode.DIRECT_REST
        self.base_url = base_url
        self.token = token

    async def execute_command(self, request: ExternalCommandRequest) -> ExternalCommandResponse:
        command_id = new_integration_id("CMD")
        endpoint = request.payload.get("endpoint")
        method = request.payload.get("method", "GET").upper()
        body: dict[str, Any] | None = request.payload.get("body")
        if not self.base_url or not endpoint:
            return ExternalCommandResponse(
                command_id=command_id,
                program_code=request.program_code,
                command_name=request.command_name,
                status=CommandStatus.ACCEPTED,
                accepted_at=datetime.utcnow(),
                adapter_mode=self.integration_mode,
                result={"dry_run": True, "required": ["base_url", "payload.endpoint"], "payload_echo": request.payload},
                message="Generic REST command accepted in dry-run mode. Configure base_url/token for live calls.",
            )
        headers = {"Authorization": f"Bearer {self.token}"} if self.token else {}
        async with httpx.AsyncClient(timeout=30.0) as client:
            response = await client.request(method, f"{self.base_url.rstrip('/')}/{endpoint.lstrip('/')}", json=body, headers=headers)
        return ExternalCommandResponse(
            command_id=command_id,
            program_code=request.program_code,
            command_name=request.command_name,
            status=CommandStatus.SUCCEEDED if response.status_code < 400 else CommandStatus.FAILED,
            accepted_at=datetime.utcnow(),
            adapter_mode=self.integration_mode,
            result={"status_code": response.status_code, "body": _safe_json(response)},
            message="Generic REST call executed.",
        )


def _safe_json(response: httpx.Response) -> Any:
    try:
        return response.json()
    except Exception:
        return response.text[:5000]

    async def execute_command(self, request: ExternalCommandRequest) -> ExternalCommandResponse:

                accepted_at=datetime.utcnow(),
                adapter_mode=self.integration_mode,
                message="Unsupported PI Web API command.",
            )
        if not self.base_url:
            return ExternalCommandResponse(
                command_id=command_id,
                program_code=request.program_code,
                command_name=request.command_name,
                status=CommandStatus.ACCEPTED,
                accepted_at=datetime.utcnow(),

                result={"dry_run": True, "pi_web_api_required": "base_url + authentication", "payload_echo": request.payload},

            )

        headers = {"Authorization": f"Bearer {self.token}"} if self.token else {}

            "get_recorded_values": f"streams/{web_id}/recorded",
            "get_interpolated_values": f"streams/{web_id}/interpolated",
            "write_value": f"streams/{web_id}/value",
            "get_asset_element": f"elements/{web_id}",
        }
        method = "POST" if request.command_name == "write_value" else "GET"
        async with httpx.AsyncClient(timeout=30.0) as client:
            response = await client.request(method, f"{self.base_url.rstrip('/')}/{path_map[request.command_name]}", headers=headers, json=request.payload.get("body"))
        return ExternalCommandResponse(
            command_id=command_id,
            program_code=request.program_code,
            command_name=request.command_name,
            status=CommandStatus.SUCCEEDED if response.status_code < 400 else CommandStatus.FAILED,

from app.domain.integration_models import ExternalCallbackRequest, ExternalCallbackResponse, ExternalCommandRequest, ExternalCommandResponse, ProgramCode, IntegrationMode


class ExternalProgramAdapter(ABC):
    """Boundary interface for external AVEVA / solver / enterprise tools.

    Direct REST APIs are used where the target product supports them. For products
    such as E3D/Hull Design where the reliable integration point is often an
    in-product plugin, PML/.NET customization, file exchange or a local agent, the
    adapter dispatches a command to the bridge/agent and waits for callback events.
    """

    program_code: ProgramCode
    integration_mode: IntegrationMode

    @abstractmethod
    async def execute_command(self, request: ExternalCommandRequest) -> ExternalCommandResponse:
        raise NotImplementedError

    async def accept_callback(self, request: ExternalCallbackRequest) -> ExternalCallbackResponse:
        raise NotImplementedError(f"Callbacks are not implemented for {self.program_code}")

    async def execute(self, request: ExternalCommandRequest) -> ExternalCommandResponse:
        adapter = self.adapters.get(request.program_code)
        if not adapter:
            return ExternalCommandResponse(
                command_id=new_integration_id("CMD"),
                program_code=request.program_code,
                command_name=request.command_name,
                status=CommandStatus.REJECTED,
                accepted_at=datetime.utcnow(),
                adapter_mode=IntegrationMode.MANUAL_REGISTRATION,
                message=f"No adapter registered for {request.program_code}",
            )
        result = await adapter.execute_command(request)
        self.commands[result.command_id] = result
        return result

    async def callback(self, program_code: ProgramCode, request: ExternalCallbackRequest) -> ExternalCallbackResponse:
        adapter = self.adapters.get(program_code)
        if adapter:
            response = await adapter.accept_callback(request)
        else:
            response = ExternalCallbackResponse(
                callback_id=new_integration_id("CB"),
                program_code=program_code,
                event_type=request.event_type,
                accepted_at=datetime.utcnow(),
                mapped_control_plane_event="generic_external_event",
                message="Callback accepted by generic fallback.",
            )
        self.callbacks[response.callback_id] = response
        return response

    def create_subscription(self, payload: IntegrationSubscriptionCreate) -> IntegrationSubscription:
        sub = IntegrationSubscription(subscription_id=new_integration_id("SUB"), **payload.model_dump())
        self.subscriptions[sub.subscription_id] = sub
        return sub

    def list_subscriptions(self) -> list[IntegrationSubscription]:
        return sorted(self.subscriptions.values(), key=lambda s: s.created_at)

# AI Gate Microsoft Suitability and Implementation Assessment

## Conclusion
Microsoft AI is suitable for the AVEVA Marine PLM AI Gate **as an enterprise AI provider**, especially when used through Azure OpenAI, Microsoft Fabric and AVEVA CONNECT industrial data context. It should not be used as an autonomous release authority.

The AI Gate must be implemented as a governed interface:

1. Microsoft AI diagnoses, explains and recommends.
2. Deterministic PLM guardrails can hold the gate even if AI suggests PASS.
3. Certified solvers such as NAPA/class tools remain the source of naval calculations.
4. Human roles such as Naval Architect, Class Authority, Engineering Approver and Principal Consultant approve release.
5. Prompt/response hash, findings, evidence references and human overrides are audited.

## Why additional development is required
The previous package had diagnostics but did not have a complete Microsoft AI interface layer. The updated package adds:

- Azure OpenAI provider adapter
- Local deterministic guardrail fallback
- AI Gate evaluation API
- AI provider capability API
- Human override API
- AI Gate SQL tables and views
- Audit hashes for prompt and response
- Findings with severity, evidence requirement and recommendation

## Environment variables

```bash
AZURE_OPENAI_ENDPOINT=https://<resource>.openai.azure.com
AZURE_OPENAI_API_KEY=<secret from Key Vault or environment>
AZURE_OPENAI_DEPLOYMENT=<deployment-name>
AZURE_OPENAI_API_VERSION=2024-10-21
```

Secrets must not be stored in SQL seed files or source code.

## Key API endpoints

```text
GET  /api/v1/ai-gates/capability
GET  /api/v1/ai-gates/providers
POST /api/v1/ai-gates/evaluate
GET  /api/v1/ai-gates/evaluations/{evaluation_id}
POST /api/v1/ai-gates/evaluations/{evaluation_id}/override
```

## Design guardrail
AI never owns certified release. The system implements the presentation principle:

> AI prioritizes, explains and proposes; certified solvers calculate; accountable humans retain release authority.

## Final update: AVEVA Industrial AI + FastAPI + SQL provider pattern

The final presentation positions AI as an AVEVA-centered industrial AI capability rather than a Microsoft-only add-on.  The final code therefore uses this provider order:

1. `AVEVA_CONNECT_INDUSTRIAL_AI` — preferred enterprise provider boundary for AVEVA CONNECT Industrial AI Assistant style diagnosis and role-gated industrial Q&A.
2. `LOCAL_RAG_OLLAMA` — on-prem Tier 1 RAG path for shipyard IP using PostgreSQL/pgvector retrieval context and Ollama-compatible local inference.
3. `AZURE_OPENAI_DEFAULT` — optional customer-tenant AI provider.
4. `LOCAL_RULES_GUARDRAIL` — deterministic FastAPI/Pydantic rule gate for fail-safe operation.

FastAPI remains the PLM control-plane authority.  PostgreSQL persists AI gate evaluations, findings, overrides, provider config, retrieval documents and retrieval chunks.  AI can only diagnose, recommend, explain and hold gates; certified solvers calculate, and accountable human/class/engineering authorities approve release.

# External Program API / Callback Integration Assessment

## Conclusion

The FastAPI control plane must not assume that every AVEVA desktop / engineering product exposes a direct public REST command API, and it must not claim that AVEVA E3D/Marine/Hull authoring runs directly on Linux. The correct production architecture is a hybrid integration gateway:

**Windows-based AVEVA engineering authoring + Linux-based Open PLM Control Plane + Cloud/CONNECT integration.**

1. **Direct REST adapter** where the product officially exposes REST APIs.
2. **Windows in-product plugin / PML.NET / local agent bridge** where the product is a desktop or engineering design environment such as E3D / Marine / Hull Design.
3. **Callback/event gateway** so external products can push design changes, drawing issues, MTO extraction, solver results, AIM document updates and PI/CONNECT operational events back into the control plane.

## Product-by-product position

| Product | Direct API command? | Callback pattern | Final design decision |
|---|---:|---:|---|
| AVEVA E3D / Marine | Not assumed as native REST in all deployments; not Linux-hosted authoring | Yes, via Windows plugin / PML.NET / local bridge | Add E3D bridge adapter and callback endpoint |
| AVEVA Hull Design | Not assumed as native REST in all deployments; not Linux-hosted authoring | Yes, via Windows bridge pattern | Add hull revision event commands and callbacks |
| AVEVA Draw | Usually via application/agent workflow | Yes | Add drawing-issued callback and output metadata capture |
| AVEVA AIM | Customer/environment-specific information interfaces | Yes | Add generic REST/OData adapter and validate endpoint contract |
| AVEVA PI System | Yes, PI Web API | Yes, polling/subscription via adapter | Add PI Web API adapter |
| AVEVA CONNECT | Yes, CONNECT data services REST APIs | Yes | Add CONNECT Data Services adapter |
| NAPA / class solvers | Usually via solver adapter / file/API bridge | Yes | Treat as solver of record and store provenance/evidence |

## Guardrail

The FastAPI API calls external tools through adapters. It should not claim that all AVEVA programs can be called directly by REST unless the specific customer-installed product version exposes and licenses that interface. For E3D/Hull, the safe and scalable pattern is: **FastAPI command on Linux → bridge queue → Windows local agent / E3D plugin / PML.NET action → callback to FastAPI**.


## Expressions to correct

| Unsafe expression | Corrected expression |
|---|---|
| AVEVA Marine / E3D on Linux server | E3D/Marine runs in supported Windows authoring environments; Linux servers run API/SQL/AI Gate/Integration Control Plane. |
| Linux server directly runs PLM and E3D | Linux runs the PLM Control Plane and integration layer; E3D is linked through Windows Authoring Client or supported AVEVA environment adapters. |
| FastAPI directly calls E3D/Hull | FastAPI issues commands; E3D/Hull Local Agent or Plugin executes and returns callbacks. |
| All AVEVA products integrated on Linux | Windows Authoring + Linux Control Plane + Cloud/CONNECT + Adapter-based Hybrid Architecture. |

# Hybrid Deployment Correction

Final deployment phrase:

**Windows-based AVEVA engineering authoring + Linux-based Open PLM Control Plane + Cloud/CONNECT integration**

## Responsibilities

| Layer | Runtime | Responsibility |
|---|---|---|
| AVEVA engineering authoring | Windows authoring clients or supported AVEVA engineering environments | E3D/Marine, Hull Design, Draw, MTO and interactive engineering work |
| Open PLM Control Plane | Linux server / container platform | FastAPI, OpenAPI, PostgreSQL, AI Gate, Evidence Store, Integration Gateway, Callback Receiver, OIDC/RBAC |
| Cloud / CONNECT / external tools | Product-specific supported environment | CONNECT, AIM, PI, NAPA/class solver adapters, ERP/MES, groupware |

## Integration rule

FastAPI does not directly manipulate AVEVA E3D/Hull internals. It issues commands to an adapter or bridge. The Windows-side plugin/local agent executes the operation and returns callbacks into the Linux control plane.

## Correct wording

- E3D/Marine runs in supported Windows authoring environments.
- Linux servers run API/SQL/AI Gate/Integration Control Plane.
- FastAPI issues commands; E3D/Hull Local Agent or Plugin executes and returns callbacks.
- This is a hybrid architecture, not a Linux-only AVEVA product deployment.

# AVEVA Marine PLM Control Plane - Final FastAPI/API/SQL Redesign

This package corrects the earlier Phase 1 / Phase 2 FastAPI code against the final presentation direction.

The final target is **not** a conventional S6 dashboard or a standalone Phase 2 BOM service. It is an **Open AVEVA Marine PLM Control Plane** that connects:

`Requirement -> VCRM/Checklist -> E3D/Hull revision -> BOM/Effectivity/Baseline -> Change Impact -> Approved Solver Evidence -> Human/Class Approval -> Operations Calibration`.

## What changed from the previous code

- Added Continuous Naval Assurance APIs.
- Added solver-run, diagnostics, evidence-pack and approval concepts.
- Reframed NAPA as `solver of record`; AVEVA owns decision/evidence/lifecycle learning.
- Reframed Siemens-grade PLM functions as the Configuration Control Plane.
- Retained `evaluate -> review -> promote` governance for official records.
- Added customer-owned Process Designer API.
- Added SQL schema for requirements, VCRM, E3D revisions, BOM, effectivity, baseline, change, solver runs, evidence, approval and operations calibration.
- Kept AI as diagnosis/recommendation/gate-hold only. AI never owns certified release authority.

### Important implementation note

Not every AVEVA engineering application should be assumed to expose a native REST command API. For E3D / Marine / Hull Design, the recommended production pattern is:

```text
FastAPI command
→ bridge queue / local agent
→ E3D or Hull in-product plugin using PML / PML.NET / .NET customization
→ callback to FastAPI
→ revision / BOM / evidence / quality gate mapping
```

Direct REST is used where officially available, such as PI Web API and CONNECT data services. AIM integration is provided through a configurable REST/OData adapter because the customer-installed endpoint contract must be validated in the target environment.

CREATE OR REPLACE VIEW vw_requirement_traceability AS
SELECT
    r.project_id,
    r.requirement_id,
    r.requirement_no,
    r.title,
    COUNT(DISTINCT v.item_id) AS vcrm_count,
    COUNT(DISTINCT c.item_id) AS checklist_count,
    COUNT(DISTINCT v.item_id) FILTER (WHERE v.status = 'APPROVED') AS approved_vcrm_count,
    COUNT(DISTINCT c.item_id) FILTER (WHERE c.status = 'APPROVED') AS approved_checklist_count
FROM plm_requirement r
LEFT JOIN plm_vcrm_item v ON r.requirement_id = v.requirement_id
LEFT JOIN plm_checklist_item c ON r.requirement_id = c.requirement_id
GROUP BY r.project_id, r.requirement_id, r.requirement_no, r.title;

CREATE OR REPLACE VIEW vw_change_assurance_status AS
SELECT
    cr.project_id,
    cr.change_request_id,
    cr.change_no,
    cr.change_type,
    cr.title,
    COUNT(DISTINCT ic.candidate_id) AS candidate_count,
    COUNT(DISTINCT oi.impact_id) AS promoted_impact_count,
    COUNT(DISTINCT sr.solver_run_id) AS solver_run_count,
    COUNT(DISTINCT ep.evidence_pack_id) AS evidence_pack_count,
    COUNT(DISTINCT ap.approval_id) FILTER (WHERE ap.decision = 'APPROVED') AS approval_count
FROM plm_change_request cr
LEFT JOIN plm_impact_candidate ic ON cr.change_request_id = ic.change_request_id
LEFT JOIN plm_official_impact oi ON cr.change_request_id = oi.change_request_id
LEFT JOIN plm_solver_run sr ON cr.change_request_id = sr.change_request_id
LEFT JOIN plm_evidence_pack ep ON ep.source_id IN (cr.change_request_id, sr.solver_run_id)
LEFT JOIN plm_approval ap ON ap.object_id IN (cr.change_request_id, ep.evidence_pack_id)
GROUP BY cr.project_id, cr.change_request_id, cr.change_no, cr.change_type, cr.title;

CREATE OR REPLACE VIEW vw_control_plane_dashboard AS
SELECT
    p.project_id,
    p.project_name,
    COUNT(DISTINCT r.requirement_id) AS requirement_count,
    COUNT(DISTINCT rev.revision_id) AS revision_event_count,
    COUNT(DISTINCT cr.change_request_id) AS change_count,
    COUNT(DISTINCT ic.candidate_id) AS impact_candidate_count,
    COUNT(DISTINCT oi.impact_id) AS official_impact_count,
    COUNT(DISTINCT sr.solver_run_id) AS solver_run_count,
    COUNT(DISTINCT ep.evidence_pack_id) AS evidence_pack_count,
    COUNT(DISTINCT ap.approval_id) FILTER (WHERE ap.decision = 'APPROVED') AS approved_count,
    COUNT(DISTINCT cal.calibration_id) AS operations_calibration_count
FROM plm_project p
LEFT JOIN plm_requirement r ON p.project_id = r.project_id
LEFT JOIN plm_e3d_revision_event rev ON p.project_id = rev.project_id
LEFT JOIN plm_change_request cr ON p.project_id = cr.project_id
LEFT JOIN plm_impact_candidate ic ON cr.change_request_id = ic.change_request_id
LEFT JOIN plm_official_impact oi ON cr.change_request_id = oi.change_request_id
LEFT JOIN plm_solver_run sr ON p.project_id = sr.project_id
LEFT JOIN plm_evidence_pack ep ON p.project_id = ep.project_id
LEFT JOIN plm_approval ap ON p.project_id = ap.project_id
LEFT JOIN plm_operations_calibration cal ON p.project_id = cal.project_id
GROUP BY p.project_id, p.project_name;

CREATE OR REPLACE FUNCTION fn_expand_bom(p_root_object_id TEXT)
RETURNS TABLE(level INTEGER, parent_object_id TEXT, child_object_id TEXT, child_object_type TEXT, quantity NUMERIC, unit TEXT, path TEXT)
LANGUAGE sql
AS $$
WITH RECURSIVE bom_tree AS (
    SELECT
        1 AS level,
        bl.parent_object_id,
        bl.child_object_id,
        bl.child_object_type,
        bl.quantity,
        bl.unit,
        bl.parent_object_id || '>' || bl.child_object_id AS path
    FROM plm_bom_line bl
    WHERE bl.parent_object_id = p_root_object_id
    UNION ALL
    SELECT
        bt.level + 1,
        bl.parent_object_id,
        bl.child_object_id,
        bl.child_object_type,
        bl.quantity,
        bl.unit,
        bt.path || '>' || bl.child_object_id AS path
    FROM plm_bom_line bl
    JOIN bom_tree bt ON bl.parent_object_id = bt.child_object_id
)
SELECT * FROM bom_tree;
$$;
